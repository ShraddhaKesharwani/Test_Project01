secrets=$(az keyvault secret show --name CONNECTIONSTRINGS-DEFAULT --vault-name $PROJECT-Vault --query 'attributes.enabled'  -o tsv)
 if [[ $secrets == "true" ]]; then
echo "secrets already uploaded"
 else
echo "add Default connection strings to Vault"
az keyvault secret set --vault-name $PROJECT-Vault  --name CONNECTIONSTRINGS--DEFAULT --value $CONSTRING 1> /dev/null
fi
